Radio & checkbox

Example

+radioCheck({
  type:"radio",
  name:'chk1'
})
+radioCheck({
  name:'chk2'
})
+radioCheck({
  type:"radio",
  name:'radiolist',
  lists:[
    {
      label:'라디오버튼1',
      disabled:true,
    },
    {
      label:'라디오버튼2'
    },
    {
      label:'라디오버튼3'
    }
  ]
})
+radioCheck({
  name:'checklist',
  lists:[
    {
      label:'체크1',
      checked:true
    },
    {
      label:'체크2'
    },
    {
      label:'체크3'
    }
  ]
})
+radioCheck({
  name:'btnType',
  type:"radio",
  className:'button_type',
  lists:[
    {
      label:'체크1'
    },
    {
      label:'체크2'
    },
    {
      label:'체크3'
    }
  ]
})


Attribute

Attributes
Property Description Type Default
type radio, checkbox string checkbox
lists 리스트 형식 데이터 array -
disabled disabled 여부 Boolean false
checked checked 여부 Boolean false
name group 이름 string radioCheck
className custom class 적용 string -
